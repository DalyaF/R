4.1 NOURRITURE

#Partie food
food<-read_delim("Production_Crops_Livestock_E_All_Data.csv",
                 locale=locale(encoding="ISO-8859-1")) |> 
  select(-ends_with("F")) |> 
  pivot_longer(cols=Y1961:Y2020,
               names_to="annee",
               values_to="Production_nourriture") |> 
  mutate(annee=parse_number(annee)) |> 
  filter(annee ==2020)
